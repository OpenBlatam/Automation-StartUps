# Security Middleware Implementation
from fastapi import FastAPI, HTTPException, Depends
from fastapi.security import HTTPBearer, HTTPAuthorizationCredentials
import jwt
import redis
from cryptography.fernet import Fernet

class SecurityManager:
    def __init__(self):
        self.redis_client = redis.Redis(host='redis-cluster', port=6379)
        self.encryption_key = Fernet.generate_key()
        self.cipher = Fernet(self.encryption_key)
    
    async def authenticate_request(self, credentials: HTTPAuthorizationCredentials):
        """Zero Trust Authentication"""
        try:
            # Verify JWT token
            payload = jwt.decode(
                credentials.credentials, 
                self.SECRET_KEY, 
                algorithms=["HS256"]
            )
            
            # Check token in Redis (for revocation)
            if not self.redis_client.exists(f"token:{payload['jti']}"):
                raise HTTPException(status_code=401, detail="Token revoked")
            
            # Verify device fingerprint
            device_fingerprint = payload.get('device_fingerprint')
            if not self.verify_device(device_fingerprint):
                raise HTTPException(status_code=401, detail="Device not trusted")
            
            return payload
            
        except jwt.ExpiredSignatureError:
            raise HTTPException(status_code=401, detail="Token expired")
        except jwt.InvalidTokenError:
            raise HTTPException(status_code=401, detail="Invalid token")
    
    def encrypt_sensitive_data(self, data: str) -> str:
        """Encrypt sensitive customer data"""
        return self.cipher.encrypt(data.encode()).decode()
    
    def decrypt_sensitive_data(self, encrypted_data: str) -> str:
        """Decrypt sensitive customer data"""
        return self.cipher.decrypt(encrypted_data.encode()).decode()

# GDPR Compliance Implementation
class DataPrivacyManager:
    def __init__(self):
        self.data_retention_policy = {
            "conversation_data": "2 years",
            "customer_personal_data": "5 years",
            "analytics_data": "1 year",
            "audit_logs": "7 years"
        }
    
    async def handle_data_subject_request(self, request_type: str, customer_id: str):
        """Handle GDPR data subject requests"""
        if request_type == "access":
            return await self.export_customer_data(customer_id)
        elif request_type == "deletion":
            return await self.delete_customer_data(customer_id)
        elif request_type == "portability":
            return await self.export_portable_data(customer_id)
        elif request_type == "rectification":
            return await self.update_customer_data(customer_id)
    
    async def anonymize_data(self, data: dict) -> dict:
        """Anonymize customer data for analytics"""
        anonymized = data.copy()
        
        # Remove PII
        pii_fields = ['email', 'phone', 'address', 'name']
        for field in pii_fields:
            if field in anonymized:
                anonymized[field] = self.hash_field(anonymized[field])
        
        # Add noise to numerical data
        if 'age' in anonymized:
            anonymized['age'] = self.add_noise(anonymized['age'], 5)
        
        return anonymized
    
    async def data_retention_cleanup(self):
        """Automated data retention cleanup"""
        for data_type, retention_period in self.data_retention_policy.items():
            cutoff_date = datetime.now() - timedelta(
                days=self.parse_retention_period(retention_period)
            )
            await self.delete_expired_data(data_type, cutoff_date)
