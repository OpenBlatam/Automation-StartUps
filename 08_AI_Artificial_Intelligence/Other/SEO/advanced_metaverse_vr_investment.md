# Virtual Real Estate Investment Analyzer
class VirtualRealEstateAnalyzer:
    def __init__(self, virtual_worlds, property_types):
        self.virtual_worlds = virtual_worlds
        self.property_types = property_types
        self.location_analyzer = VirtualLocationAnalyzer()
        self.demand_forecaster = VirtualDemandForecaster()
        self.value_estimator = VirtualValueEstimator()
    
    def analyze_virtual_property(self, virtual_world, property_location, property_type):
        # Analyze virtual location
        location_analysis = self.location_analyzer.analyze_location(
            virtual_world, property_location
        )
        
        # Forecast demand
        demand_forecast = self.demand_forecaster.forecast_demand(
            virtual_world, property_location, property_type
        )
        
        # Estimate property value
        property_value = self.value_estimator.estimate_value(
            virtual_world, property_location, property_type, location_analysis, demand_forecast
        )
        
        # Calculate investment potential
        investment_potential = self.calculate_investment_potential(
            property_value, demand_forecast, location_analysis
        )
        
        return VirtualRealEstateAnalysis(
            location_analysis, demand_forecast, property_value, investment_potential
        )
    
    def calculate_investment_potential(self, value, demand, location):
        # Calculate ROI based on value, demand, and location
        roi = (demand * location.score * value.growth_rate) / value.current_price
        return InvestmentPotential(roi, value, demand, location)
