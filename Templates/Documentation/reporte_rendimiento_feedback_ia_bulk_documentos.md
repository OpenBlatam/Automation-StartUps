def calcular_ahorro_tiempo(documentos, tiempo_manual, tiempo_ia):
    """Calcula ahorro de tiempo total"""
    tiempo_total_manual = documentos * tiempo_manual
    tiempo_total_ia = tiempo_ia  # Tiempo constante independiente de volumen

        'tiempo_manual': tiempo_total_manual,
        'tiempo_ia': tiempo_total_ia,
        'ahorro': ahorro,
        'porcentaje_ahorro': porcentaje_ahorro
    }
