```python
def calcular_puntos_documento(tipo_documento, complejidad, detalles):
    """
    Calcula UP, QP y EP basado en tipo y calidad de documento
    """
    # Puntos de Uso (UP)
    up_base = {
        'email': 50,
        'reporte_basico': 75,
        'propuesta_comercial': 150,
        'documento_tecnico': 200,
        'contrato_legal': 175,
        'presentacion_ejecutiva': 125
    }
    
    up = up_base.get(tipo_documento, 50)
    
    # Bonos UP
    if detalles.get('consulta_unica'):
        up += 50
    if detalles.get('multi_formato'):
        up += 30
    if detalles.get('reutiliza_template'):
        up = 75  # Override para reutilizaci√≥n
    
    # Puntos de Calidad (QP)
    qp = 0
    if detalles.get('aprobado_sin_revisiones'):
        qp = 150
    if detalles.get('usado_como_template'):
        qp = 200
    if detalles.get('primera_version_perfecta'):
        qp = 250
    if detalles.get('mejor_practica'):
        qp = 300
    
    # Puntos de Eficiencia (EP)
    ep = 0
    tiempo_ahorrado_pct = detalles.get('tiempo_ahorrado_pct', 0)
    if tiempo_ahorrado_pct >= 90:
        ep = 500
    elif tiempo_ahorrado_pct >= 70:
        ep = 200
    elif tiempo_ahorrado_pct >= 50:
        ep = 100
    
    if detalles.get('automatizacion_completa'):
        ep += 300
    
    return {'up': up, 'qp': qp, 'ep': ep}

def calcular_tiempo_ahorrado(tiempo_tradicional, tiempo_ia_bulk):
    """
    Calcula porcentaje de tiempo ahorrado
    """
    if tiempo_tradicional == 0:
        return 0
    return ((tiempo_tradicional - tiempo_ia_bulk) / tiempo_tradicional) * 100
```

```sql
-- Query para dashboard de productividad
SELECT 
    u.nombre,
    u.departamento,
    COUNT(d.id) as documentos_generados,
    SUM(d.up) as up_total,
    SUM(d.qp) as qp_total,
    SUM(d.ep) as ep_total,
    AVG(d.tiempo_ahorrado_pct) as tiempo_ahorrado_promedio,
    COUNT(CASE WHEN d.aprobado_sin_revisiones = 1 THEN 1 END) * 100.0 / 
        COUNT(d.id) as tasa_aprobacion_primera_version,
    u.nivel_actual
FROM usuarios u
LEFT JOIN documentos d ON u.id = d.usuario_id
WHERE d.fecha >= DATE_SUB(NOW(), INTERVAL 7 DAY)
GROUP BY u.id
ORDER BY (up_total + qp_total + ep_total) DESC
LIMIT 10;

-- Query para an√°lisis de templates
SELECT 
    t.nombre,
    t.categoria,
    COUNT(d.id) as veces_usado,
    AVG(d.qp) as calidad_promedio,
    AVG(d.tiempo_ahorrado_pct) as eficiencia_promedio,
    AVG(t.rating) as rating_promedio
FROM templates t
LEFT JOIN documentos d ON t.id = d.template_id
GROUP BY t.id
ORDER BY veces_usado DESC
LIMIT 20;
```
