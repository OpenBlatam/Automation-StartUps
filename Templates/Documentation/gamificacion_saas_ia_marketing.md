```python
        vp_base = int(vp_base * 0.6)  # 60% de puntos por renovaci√≥n
    elif tipo_venta == 'upsell':
        vp_base = int(vp_base * 0.4)  # 40% por upsell
    
    return vp_base + bonos

def calcular_ap(actividad_adopcion, detalles):
    """
    Calcula AP (Puntos de Adopci√≥n) basado en actividad
    """
    ap_base = {
        'onboarding': 200,
        'cliente_activo': 100,
        'funcionalidad_avanzada': 150,
        'caso_exito': 300,
        'retencion': 250,
        'expansion_uso': 200
    }
    
    ap = ap_base.get(actividad_adopcion, 0)
    
    # Bonos
    if actividad_adopcion == 'onboarding' and detalles.get('rapido'):  # < 14 d√≠as
        ap += 100
    if actividad_adopcion == 'caso_exito' and detalles.get('publicado'):
        ap += 200
    
    return ap
```

```sql
-- Query para dashboard de ventas semanal
SELECT 
    v.nombre,
    v.region,
    SUM(vp.puntos) as vp_semana,
    SUM(ap.puntos) as ap_semana,
    COUNT(DISTINCT vp.venta_id) as ventas_cerradas,
    SUM(v.valor_venta) as mrr_generado,
    v.nivel_actual,
    v.bonos_acumulados
FROM vendedores v

LEFT JOIN puntos_adopcion ap ON v.id = ap.vendedor_id

   OR ap.fecha >= DATE_SUB(NOW(), INTERVAL 7 DAY)
GROUP BY v.id
ORDER BY (vp_semana + ap_semana) DESC
LIMIT 10;

-- Query para tasa de cierre
SELECT 
    COUNT(DISTINCT CASE WHEN o.estado = 'Cerrada' THEN o.id END) * 100.0 / 
    COUNT(DISTINCT o.id) as tasa_cierre,
    AVG(DATEDIFF(o.fecha_cierre, o.fecha_creacion)) as ciclo_promedio
FROM oportunidades o
WHERE o.fecha_creacion >= DATE_SUB(NOW(), INTERVAL 30 DAY);
```
