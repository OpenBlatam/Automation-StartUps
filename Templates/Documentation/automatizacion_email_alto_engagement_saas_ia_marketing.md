```javascript
// Script para tracking de eventos del producto
class ProductEventTracker {
    constructor(apiEndpoint, apiKey) {
        this.apiEndpoint = apiEndpoint;
        this.apiKey = apiKey;
    }
    
    trackEvent(userId, eventType, eventData) {
        const events = {
            'login': { points: 5, weight: 1 },
            'campaign_create': { points: 15, weight: 2 },
            'campaign_activate': { points: 20, weight: 3 },
            'feature_use': { points: 20, weight: 2 },
            'integration_connect': { points: 25, weight: 3 },
            'roi_generated': { points: 50, weight: 5 },
            'team_invite': { points: 30, weight: 3 }
        };
        
        const event = events[eventType];
        if (!event) return;
        
        // Calcular puntos con peso
        const points = event.points * event.weight;
        
        // Enviar a API
        fetch(`${this.apiEndpoint}/events`, {
            method: 'POST',
            headers: {
                'Authorization': `Bearer ${this.apiKey}`,
                'Content-Type': 'application/json'
            },
            body: JSON.stringify({
                user_id: userId,
                event_type: eventType,
                points: points,
                data: eventData,
                timestamp: new Date().toISOString()
            })
        })
        .then(response => response.json())
        .then(data => {
            // Actualizar engagement score
            this.updateEngagementScore(userId, points);
            
            // Trigger workflows si es necesario
            this.checkWorkflowTriggers(userId, eventType);
        });
    }
    
    updateEngagementScore(userId, points) {
        // Obtener score actual
        fetch(`${this.apiEndpoint}/users/${userId}/engagement`, {
            headers: {
                'Authorization': `Bearer ${this.apiKey}`
            }
        })
        .then(response => response.json())
        .then(data => {
            const newScore = data.score + points;
            
            // Actualizar score
            return fetch(`${this.apiEndpoint}/users/${userId}/engagement`, {
                method: 'PUT',
                headers: {
                    'Authorization': `Bearer ${this.apiKey}`,
                    'Content-Type': 'application/json'
                },
                body: JSON.stringify({ score: newScore })
            });
        })
        .then(() => {
            // Actualizar segmento
            this.updateSegment(userId, newScore);
        });
    }
    
    checkWorkflowTriggers(userId, eventType) {
        // Verificar si evento activa alg√∫n workflow
        const triggers = {
            'roi_generated': 'power_user_celebration',
            'campaign_create': 'campaign_tips',
            'feature_use': 'feature_optimization'
        };
        
        if (triggers[eventType]) {
            this.triggerWorkflow(userId, triggers[eventType]);
        }
    }
}
```

```python
    def __init__(self, user_data, feature_catalog):
        self.user_data = user_data
        self.feature_catalog = feature_catalog
    
    def recommend_features(self, top_n=3):
        """
        Recomienda features basado en uso y comportamiento
        """
        # Analizar uso actual
        used_features = self.user_data['features_used']
        available_features = [f for f in self.feature_catalog if f not in used_features]
        
        # Analizar tipo de campa√±as
        campaign_types = self.analyze_campaign_types()
        
        # Calcular relevancia de cada feature
        recommendations = []
        for feature in available_features:
            relevance_score = self.calculate_relevance(feature, campaign_types)
            roi_potential = self.calculate_roi_potential(feature)
            ease_of_use = feature['setup_time']  # Menor = mejor
            
            # Score combinado
            combined_score = (relevance_score * 0.4) + (roi_potential * 0.4) + ((100 - ease_of_use) * 0.2)
            
            recommendations.append({
                'feature': feature,
                'relevance': relevance_score,
                'roi_potential': roi_potential,
                'ease_of_use': ease_of_use,
                'combined_score': combined_score
            })
        
        # Ordenar y retornar top N
        recommendations.sort(key=lambda x: x['combined_score'], reverse=True)
        return recommendations[:top_n]
    
    def calculate_relevance(self, feature, campaign_types):
        """
        Calcula relevancia de feature basado en tipo de campa√±as
        """
        relevance_map = {
            'email_campaigns': ['email_personalization', 'ab_testing', 'email_automation'],
            'social_campaigns': ['social_scheduling', 'content_creation', 'analytics'],
            'paid_campaigns': ['ad_optimization', 'budget_management', 'conversion_tracking']
        }
        
        relevance = 0
        for campaign_type in campaign_types:
            if feature['name'] in relevance_map.get(campaign_type, []):
                relevance += 30
        
        return min(relevance, 100)
    
    def calculate_roi_potential(self, feature):
        """
        Calcula ROI potencial basado en usuarios similares
        """
        similar_users = self.find_similar_users()
        roi_improvements = [u['roi_improvement'] for u in similar_users if feature['name'] in u['features_used']]
        
        if roi_improvements:
            return sum(roi_improvements) / len(roi_improvements)
        return 0
```

```python
    risk_factors = []
    risk_score = 0
    
    # Factor 1: Decremento en login frequency
    if user.login_frequency_trend < -0.3:  # -30% o m√°s

        risk_score += 30
    
    # Factor 2: Sin actividad reciente
    days_since_login = (datetime.now() - user.last_login).days
    if days_since_login > 14:
        risk_factors.append('inactive')
        risk_score += 25
    
    # Factor 3: ROI bajo o decreciente
    if user.roi < 5 or user.roi_trend < -0.2:
        risk_factors.append('low_roi')
        risk_score += 20
    
    # Factor 4: No usa features avanzadas
    if user.feature_usage_rate < 0.3:
        risk_factors.append('low_feature_usage')
        risk_score += 15
    
    # Factor 5: Alcanz√≥ l√≠mites pero no upgrade
    if user.plan_limits_reached and user.plan != 'pro':
        risk_factors.append('limits_reached')
        risk_score += 10
    
    # Determinar nivel de riesgo
    if risk_score >= 50:
        risk_level = 'high'
    elif risk_score >= 30:
        risk_level = 'medium'
    else:
        risk_level = 'low'
    
    return {
```

```sql
-- An√°lisis de adopci√≥n de features por segmento
SELECT 
    CASE 
        WHEN engagement_score >= 300 THEN 'Power User'
        WHEN engagement_score >= 200 THEN 'Active User'
        WHEN engagement_score >= 100 THEN 'Regular User'
        ELSE 'Casual User'
    END AS segment,
    COUNT(DISTINCT u.id) AS total_users,
    COUNT(DISTINCT f.user_id) AS users_with_features,
    COUNT(DISTINCT f.feature_name) AS unique_features_used,
    AVG(feature_count) AS avg_features_per_user,
    COUNT(DISTINCT f.feature_name) * 100.0 / 
        (SELECT COUNT(*) FROM features_catalog) AS adoption_rate
FROM users u
LEFT JOIN (
    SELECT user_id, feature_name, COUNT(*) as feature_count
    FROM feature_usage
    WHERE used_at >= DATE_SUB(NOW(), INTERVAL 30 DAY)
    GROUP BY user_id, feature_name
) f ON u.id = f.user_id
WHERE u.created_at >= DATE_SUB(NOW(), INTERVAL 90 DAY)
GROUP BY segment
ORDER BY avg_features_per_user DESC;
```

### Query 2: ROI por Feature

```sql
-- Calcular ROI de cada feature
SELECT 
    f.feature_name,
    COUNT(DISTINCT f.user_id) AS users_with_feature,
    AVG(u.roi) AS avg_roi_with_feature,
    (SELECT AVG(roi) FROM users u2 
     WHERE u2.id NOT IN (SELECT user_id FROM feature_usage WHERE feature_name = f.feature_name)
     AND u2.created_at >= DATE_SUB(NOW(), INTERVAL 90 DAY)) AS avg_roi_without_feature,
    AVG(u.roi) - (SELECT AVG(roi) FROM users u2 
                   WHERE u2.id NOT IN (SELECT user_id FROM feature_usage WHERE feature_name = f.feature_name)
                   AND u2.created_at >= DATE_SUB(NOW(), INTERVAL 90 DAY)) AS roi_improvement,
    AVG(u.revenue) AS avg_revenue_with_feature,
    (SELECT AVG(revenue) FROM users u2 
     WHERE u2.id NOT IN (SELECT user_id FROM feature_usage WHERE feature_name = f.feature_name)
     AND u2.created_at >= DATE_SUB(NOW(), INTERVAL 90 DAY)) AS avg_revenue_without_feature
FROM feature_usage f
JOIN users u ON f.user_id = u.id
WHERE f.used_at >= DATE_SUB(NOW(), INTERVAL 90 DAY)
GROUP BY f.feature_name
HAVING users_with_feature >= 10
ORDER BY roi_improvement DESC;
```
