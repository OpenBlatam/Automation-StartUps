class OptimizationRecommendationEngine:
    def __init__(self, user_data, optimization_catalog):
        self.user_data = user_data
        self.optimization_catalog = optimization_catalog
    
    def recommend_optimizations(self, top_n=3):
        """
        Recomienda optimizaciones basado en patrones de procesamiento
        """
        # Analizar patrones actuales
        processing_patterns = self.analyze_processing_patterns()
        document_types = self.analyze_document_types()
        
        # Calcular relevancia de cada optimizaciÃ³n
        recommendations = []
        for optimization in self.optimization_catalog:
            relevance = self.calculate_relevance(optimization, processing_patterns, document_types)
            time_saved = self.calculate_time_saved(optimization)
            cost_saved = time_saved * self.user_data['hourly_rate']
            ease_of_implementation = optimization['setup_time']
            
            # Score combinado
            combined_score = (relevance * 0.3) + (time_saved * 0.3) + (cost_saved / 100 * 0.2) + ((100 - ease_of_implementation) * 0.2)
            
            recommendations.append({
                'optimization': optimization,
                'relevance': relevance,
                'time_saved': time_saved,
                'cost_saved': cost_saved,
                'ease_of_implementation': ease_of_implementation,
                'combined_score': combined_score
            })
        
        recommendations.sort(key=lambda x: x['combined_score'], reverse=True)
        return recommendations[:top_n]
    
    def calculate_relevance(self, optimization, patterns, doc_types):
        """
        Calcula relevancia basado en patrones
        """
        relevance = 0
        
        # Si procesa individualmente pero optimizaciÃ³n es para lotes

            relevance += 40
        
        # Si procesa mÃºltiples tipos pero no usa templates

            relevance += 35
        
        # Si tiene documentos urgentes pero no usa prioridad
        if patterns['urgent_documents'] > 0.2 and optimization['type'] == 'priority_processing':

        return min(relevance, 100)

-- Calcular ROI de optimizaciones
SELECT 
    o.optimization_type,
    COUNT(DISTINCT o.user_id) AS users_with_optimization,
    AVG(u.time_saved_monthly) AS avg_time_saved,
    AVG(u.time_saved_monthly * u.hourly_rate) AS avg_cost_saved,
    AVG(o.setup_time) AS avg_setup_time,
    AVG(o.setup_time * 50) AS avg_setup_cost,

    AVG(u.efficiency_score) AS avg_efficiency_improvement

JOIN users u ON o.user_id = u.id
WHERE o.implemented_at >= DATE_SUB(NOW(), INTERVAL 90 DAY)
GROUP BY o.optimization_type
HAVING users_with_optimization >= 5
ORDER BY roi DESC;
