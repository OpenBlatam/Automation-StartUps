def verificar_nivel(xp_total):
    """
    Verifica y actualiza nivel basado en XP total

    niveles = [

        (501, 1500, 2, "Aprendiz"),
        (1501, 3000, 3, "Intermedio"),
        (3001, 5000, 4, "Avanzado"),
        (5001, float('inf'), 5, "Experto")
    ]
    
    for min_xp, max_xp, nivel_id, nombre in niveles:
        if min_xp <= xp_total <= max_xp:

def identificar_riesgo_abandono():
    """
    Identifica usuarios en riesgo de abandono
    """
    usuarios = obtener_usuarios_activos()
    usuarios_riesgo = []
    
    for usuario in usuarios:
        if usuario.dias_sin_actividad > 7:
            engagement = analizar_engagement(usuario.id)
            if engagement['riesgo_abandono'] == 'alto':
                usuarios_riesgo.append({
                    'usuario': usuario,
                    'engagement': engagement,
                    'accion_recomendada': engagement['recomendaciones']
                })
    
    return usuarios_riesgo
