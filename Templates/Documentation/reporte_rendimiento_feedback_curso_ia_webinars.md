## üí∞ Calculadora de ROI Detallada

### F√≥rmula de ROI para Curso

```
ROI = ((Ingresos - Costos) / Costos) √ó 100

Donde:
Ingresos = 
  (Inscripciones √ó Precio Curso) +
  (Upsells √ó Precio Upsell) +
  (Referidos √ó Comisi√≥n Referido) +
  (Renovaciones √ó Precio Renovaci√≥n)

Costos = 
  (Costo Creaci√≥n Contenido) +
  (Costo Plataforma) +
  (Costo Marketing) +
  (Costo Instructores) +
  (Costo Soporte) +
  (Costo Infraestructura)
```

### Ejemplo de C√°lculo

**Escenario Base:**
- Inscripciones: 150 estudiantes
- Precio curso: $497
- Upsells: 30 estudiantes √ó $197 = $5,910
- Referidos: 25 estudiantes √ó $50 comisi√≥n = $1,250

```
ROI = (($81,710 - $41,500) / $41,500) √ó 100
ROI = ($40,210 / $41,500) √ó 100
ROI = 96.9%
```

**Payback Period:**
```
Payback = Costos / (Ingresos / Meses)
```

```python
def calcular_tendencia(serie_temporal):
    """Calcula tendencia de una serie temporal"""
    from scipy import stats
    
    x = range(len(serie_temporal))
    slope, intercept, r_value, p_value, std_err = stats.linregress(x, serie_temporal)
    
    return {
        'tendencia': 'creciente' if slope > 0 else 'decreciente',
        'pendiente': slope,
        'r_cuadrado': r_value**2,
        'significativo': p_value < 0.05
    }
```

### Detector de Anomal√≠as

```python
def detectar_anomalias(metricas):
    """Detecta anomal√≠as en m√©tricas"""
    import numpy as np
    
    media = np.mean(metricas)
    std = np.std(metricas)
    
    anomalias = []
    for i, valor in enumerate(metricas):
        z_score = (valor - media) / std
        if abs(z_score) > 2:  # M√°s de 2 desviaciones est√°ndar
            anomalias.append({
                'indice': i,
                'valor': valor,
                'z_score': z_score,
                'tipo': 'alto' if z_score > 0 else 'bajo'
            })
    
    return anomalias
```

```python
def calcular_tamano_muestra(poblacion, nivel_confianza=0.95, margen_error=0.05):
    """Calcula tama√±o de muestra necesario para encuesta"""
    from scipy.stats import norm
    
    z_score = norm.ppf((1 + nivel_confianza) / 2)
    p = 0.5  # Asumir m√°xima variabilidad
    
    n = (z_score**2 * p * (1 - p)) / (margen_error**2)
    n_ajustado = n / (1 + (n - 1) / poblacion)
    
    return int(n_ajustado)

# Ejemplo: Curso con 500 estudiantes, 95% confianza, 5% margen error
tamano = calcular_tamano_muestra(500, 0.95, 0.05)
print(f"Tama√±o de muestra necesario: {tamano} estudiantes")
```

```python
    """Test de significancia entre dos grupos"""
    from scipy.stats import ttest_ind
    
    estadistico, p_value = ttest_ind(grupo_a, grupo_b)
    
    return {
```

```python
from transformers import pipeline
import pandas as pd

class AnalizadorSentimientoAvanzado:
    def __init__(self):
        self.sentiment_analyzer = pipeline(
            "sentiment-analysis",
            model="nlptown/bert-base-multilingual-uncased-sentiment"
        )
        self.emotion_analyzer = pipeline(
            "text-classification",
            model="j-hartmann/emotion-english-distilroberta-base"
        )
    
    def analizar_feedback(self, feedback_text):
        """An√°lisis completo de sentimiento y emociones"""
        resultado = {
            'sentimiento': self.sentiment_analyzer(feedback_text)[0],
            'emocion': self.emotion_analyzer(feedback_text)[0],
            'puntuacion': self._calcular_puntuacion(feedback_text)
        }
        return resultado
    
    def _calcular_puntuacion(self, texto):
        """Calcula puntuaci√≥n num√©rica de sentimiento"""
        sent = self.sentiment_analyzer(texto)[0]
        if 'POSITIVE' in sent['label']:
            return sent['score']
        else:
            return -sent['score']
    
    def analizar_lote(self, feedbacks_df):
        """An√°lisis de lote de feedbacks"""
        resultados = []
        for feedback in feedbacks_df['comentario']:
            resultado = self.analizar_feedback(feedback)
            resultados.append(resultado)
        return pd.DataFrame(resultados)
```

```python
from sklearn.decomposition import LatentDirichletAllocation
from sklearn.feature_extraction.text import CountVectorizer
import pandas as pd

class AnalizadorTopicos:
    def __init__(self, n_topics=5):
        self.n_topics = n_topics
        self.vectorizer = CountVectorizer(max_features=100, stop_words='spanish')
        self.lda = LatentDirichletAllocation(n_components=n_topics, random_state=42)
    
    def identificar_topicos(self, textos):
        """Identifica t√≥picos principales en feedback"""
        # Vectorizar textos
        X = self.vectorizer.fit_transform(textos)
        
        # Aplicar LDA
        self.lda.fit(X)
        
        # Obtener t√≥picos
        topicos = []
        feature_names = self.vectorizer.get_feature_names_out()
        
        for topic_idx, topic in enumerate(self.lda.components_):
            top_words = [feature_names[i] for i in topic.argsort()[-10:][::-1]]
            topicos.append({
                'topico': topic_idx,
                'palabras_clave': top_words,
                'peso': topic.sum()
            })
        
        return topicos
    
    def clasificar_feedback(self, textos):
        """Clasifica feedback por t√≥pico"""
        X = self.vectorizer.transform(textos)
        distribuciones = self.lda.transform(X)
        return distribuciones

# Uso
```

```python
import networkx as nx
import pandas as pd

class AnalizadorRedesSociales:
    def __init__(self):
        self.G = nx.Graph()
    
    def construir_red(self, interacciones_df):
        """Construye red de interacciones entre estudiantes"""
        for _, row in interacciones_df.iterrows():
            estudiante_a = row['estudiante_a']
            estudiante_b = row['estudiante_b']
            peso = row['interacciones']
            
            if self.G.has_edge(estudiante_a, estudiante_b):
                self.G[estudiante_a][estudiante_b]['weight'] += peso
            else:
                self.G.add_edge(estudiante_a, estudiante_b, weight=peso)
        
        return self.G
    
    def identificar_influencers(self, top_n=10):
        """Identifica estudiantes m√°s influyentes"""
        centralidad = nx.degree_centrality(self.G)
        influencers = sorted(
            centralidad.items(),
            key=lambda x: x[1],
            reverse=True
        )[:top_n]
        return influencers
    
    def identificar_comunidades(self):
        """Identifica comunidades de estudiantes"""
        comunidades = nx.community.greedy_modularity_communities(self.G)
        return comunidades
    
    def analizar_impacto_red(self, estudiante):
        """Analiza impacto de estudiante en la red"""
        return {
            'grado': self.G.degree(estudiante),
            'centralidad': nx.degree_centrality(self.G)[estudiante],
            'betweenness': nx.betweenness_centrality(self.G)[estudiante],
            'clustering': nx.clustering(self.G, estudiante)
        }

# Uso
```
