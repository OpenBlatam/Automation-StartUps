"""
calculate_compensation.py
Calcula compensaci√≥n autom√°tica basada en downtime y clientes afectados
"""

import csv
from datetime import datetime, timedelta

def calculate_compensation(downtime_minutes, customers_affected, avg_price):
    """Calcula compensaci√≥n basada en severidad"""
    
    # Factores de compensaci√≥n seg√∫n tiempo
    if downtime_minutes < 30:
        factor = 0.1  # 10%
    elif downtime_minutes < 120:
        factor = 0.25  # 25%
    elif downtime_minutes < 360:
        factor = 0.5  # 50%
    else:
        factor = 1.0  # 100% + adicional
    
    base_compensation = customers_affected * avg_price * factor
    
    # Cr√©dito adicional para incidentes > 6 horas
    if downtime_minutes > 360:
        additional = customers_affected * avg_price * 0.25
    else:
        additional = 0
    
    total = base_compensation + additional
    
    return {
        'base_compensation': base_compensation,
        'additional': additional,
        'total': total,
        'factor': factor,
        'per_customer': total / customers_affected if customers_affected > 0 else 0
    }

# Ejemplo de uso
if __name__ == "__main__":
    result = calculate_compensation(
        downtime_minutes=180,  # 3 horas
        customers_affected=500,
        avg_price=297
    )

    print(f"Por Cliente: ${result['per_customer']:,.2f}")
