import csv
from datetime import datetime, timedelta

def calculate_compensation(downtime_minutes, customers_affected, avg_price):
    """Calcula compensaci√≥n basada en severidad"""
    
    # Factores de compensaci√≥n seg√∫n tiempo
    if downtime_minutes < 30:
        factor = 0.1  # 10%
    elif downtime_minutes < 120:
        factor = 0.25  # 25%
    elif downtime_minutes < 360:
        factor = 0.5  # 50%
    else:
        factor = 1.0  # 100% + adicional
    
    base_compensation = customers_affected * avg_price * factor
    
    # Cr√©dito adicional para incidentes > 6 horas
    if downtime_minutes > 360:
        additional = customers_affected * avg_price * 0.25
    else:
        additional = 0
    
    total = base_compensation + additional
    
    return {
        'base_compensation': base_compensation,
        'additional': additional,
        'total': total,
        'factor': factor,
        'per_customer': total / customers_affected if customers_affected > 0 else 0
    }

# Ejemplo de uso

# auto_recover_webinar.sh
# Monitorea y recupera autom√°ticamente servicios de webinar

PRIMARY_PLATFORM="zoom"

    local platform=$1
    case $platform in
        zoom)
            # Verificar API de Zoom
            response=$(curl -s -o /dev/null -w "%{http_code}" \
                "https://api.zoom.us/v2/users/me" \
                -H "Authorization: Bearer $ZOOM_TOKEN")
            [ "$response" = "200" ] && return 0 || return 1
            ;;
        teams)
            # Verificar Microsoft Graph API
            response=$(curl -s -o /dev/null -w "%{http_code}" \
                "https://graph.microsoft.com/v1.0/me" \
                -H "Authorization: Bearer $TEAMS_TOKEN")
            [ "$response" = "200" ] && return 0 || return 1
            ;;
    esac
}

# Check primary platform
if ! check_platform "$PRIMARY_PLATFORM"; then
    echo "ALERT: Primary platform ($PRIMARY_PLATFORM) is down!"
    
    # Notify team
    echo "Primary webinar platform failure detected. Switching to backup." | \
        mail -s "URGENT: Platform Switch" "$NOTIFICATION_EMAIL"
    
    # Activate backup
    if check_platform "$BACKUP_PLATFORM"; then
        # Generate new link
        NEW_LINK=$(generate_backup_link "$WEBINAR_ID" "$BACKUP_PLATFORM")
        
        # Email participants
        send_bulk_email "$WEBINAR_ID" "New webinar link: $NEW_LINK"
        
        # Log incident
        log_incident "$WEBINAR_ID" "$PRIMARY_PLATFORM" "$BACKUP_PLATFORM"
        
        echo "Successfully switched to $BACKUP_PLATFORM"
        exit 0
    else
        echo "ERROR: Both platforms are down!"
        send_critical_alert
        exit 1
    fi
fi

echo "All systems operational"

### 31.2 Health Check y Auto-Restart

```python
#!/usr/bin/env python3
"""
health_check_auto_restart.py
Verifica salud de servicios y reinicia si es necesario
"""

import requests
import subprocess
import time
from datetime import datetime

SERVICES = {
    'web': {'url': 'https://cursos.example.com', 'port': 80},
    'api': {'url': 'https://api.example.com/health', 'port': 443},
    'database': {'host': 'db.example.com', 'port': 5432}
}

MAX_RETRIES = 3
RESTART_COMMANDS = {
    'web': 'sudo systemctl restart nginx',
    'api': 'sudo systemctl restart api-service',
    'database': 'sudo systemctl restart postgresql'
}

def check_service(service_name, config):
    """Verificar si servicio est√° saludable"""
    try:
        if 'url' in config:
            response = requests.get(config['url'], timeout=5)
            return response.status_code == 200
        elif 'host' in config:
            # Check database connection
            import socket
            sock = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
            result = sock.connect_ex((config['host'], config['port']))
            sock.close()
            return result == 0
    except:
        return False

def restart_service(service_name):
    """Reiniciar servicio"""
    if service_name in RESTART_COMMANDS:
        try:
            subprocess.run(RESTART_COMMANDS[service_name].split(), 
                         check=True, capture_output=True)
            return True
        except subprocess.CalledProcessError as e:
            print(f"Error restarting {service_name}: {e}")
            return False
    return False

def main():
    """Monitoreo continuo"""
    failures = {service: 0 for service in SERVICES}
    
    while True:
        for service, config in SERVICES.items():
            if not check_service(service, config):
                failures[service] += 1
                print(f"[{datetime.now()}] {service} failed ({failures[service]} times)")
                
                if failures[service] >= MAX_RETRIES:
                    print(f"[{datetime.now()}] Restarting {service}...")
                    if restart_service(service):
                        print(f"[{datetime.now()}] {service} restarted successfully")
                        failures[service] = 0
                        # Notify team
                        send_alert(f"{service} was restarted after {MAX_RETRIES} failures")
                    else:
                        send_critical_alert(f"{service} failed to restart!")
            else:
                failures[service] = 0
        
        time.sleep(60)  # Check every minute

if __name__ == "__main__":
    main()
```
