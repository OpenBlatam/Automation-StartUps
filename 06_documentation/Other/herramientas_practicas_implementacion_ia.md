# Sistema de A/B Testing
import numpy as np
from scipy import stats
import pandas as pd
from datetime import datetime, timedelta

class ABTestingEngine:
    def __init__(self):
        self.experiments = {}
        self.results = {}
        self.statistical_significance = 0.95
    
    def create_experiment(self, name, hypothesis, variants, traffic_split=0.5):
        """Crear experimento A/B"""
        experiment = {
            'name': name,
            'hypothesis': hypothesis,
            'variants': variants,
            'traffic_split': traffic_split,
            'start_date': datetime.now(),
            'status': 'running',
            'results': {}
        }
        
        self.experiments[name] = experiment
        return experiment
    
    def assign_variant(self, user_id, experiment_name):
        """Asignar variante a usuario"""
        if experiment_name not in self.experiments:
            return None
        
        experiment = self.experiments[experiment_name]
        
        # Asignar basado en hash del user_id
        hash_value = hash(user_id) % 100
        if hash_value < experiment['traffic_split'] * 100:
            return experiment['variants'][0]
        else:
            return experiment['variants'][1]
    
    def track_conversion(self, user_id, experiment_name, converted):
        """Rastrear conversi√≥n"""
        if experiment_name not in self.experiments:
            return False
        
        experiment = self.experiments[experiment_name]
        variant = self.assign_variant(user_id, experiment_name)
        
        if variant not in experiment['results']:
            experiment['results'][variant] = {
                'visitors': 0,
                'conversions': 0,
                'conversion_rate': 0
            }
        
        experiment['results'][variant]['visitors'] += 1
        if converted:
            experiment['results'][variant]['conversions'] += 1
        
        experiment['results'][variant]['conversion_rate'] = (
            experiment['results'][variant]['conversions'] / 
            experiment['results'][variant]['visitors']
        )
        
        return True
    
    def calculate_statistical_significance(self, experiment_name):
        """Calcular significancia estad√≠stica"""
        if experiment_name not in self.experiments:
            return None
        
        experiment = self.experiments[experiment_name]
        results = experiment['results']
        
        if len(results) != 2:
            return None
        
        variants = list(results.keys())
        variant_a = results[variants[0]]
        variant_b = results[variants[1]]
        
        # Calcular significancia estad√≠stica
        n_a = variant_a['visitors']
        n_b = variant_b['visitors']
        p_a = variant_a['conversion_rate']
        p_b = variant_b['conversion_rate']
        
        # Test de proporciones
        z_score, p_value = self.proportion_test(n_a, n_b, p_a, p_b)
        
        return {
            'z_score': z_score,
            'p_value': p_value,
            'significant': p_value < (1 - self.statistical_significance),
            'confidence_level': self.statistical_significance
        }
    
    def proportion_test(self, n_a, n_b, p_a, p_b):
        """Test de proporciones"""
        # Calcular proporci√≥n combinada
        p_combined = (n_a * p_a + n_b * p_b) / (n_a + n_b)
        
        # Calcular error est√°ndar
        se = np.sqrt(p_combined * (1 - p_combined) * (1/n_a + 1/n_b))
        
        # Calcular z-score
        z_score = (p_a - p_b) / se
        
        # Calcular p-value
        p_value = 2 * (1 - stats.norm.cdf(abs(z_score)))
        
        return z_score, p_value
    
    def get_experiment_results(self, experiment_name):
        """Obtener resultados del experimento"""
        if experiment_name not in self.experiments:
            return None
        
        experiment = self.experiments[experiment_name]
        significance = self.calculate_statistical_significance(experiment_name)
        
        return {
            'experiment': experiment,
            'significance': significance,
            'recommendation': self.get_recommendation(experiment, significance)
        }
    
    def get_recommendation(self, experiment, significance):
        """Obtener recomendaci√≥n basada en resultados"""
        if not significance or not significance['significant']:
            return "Continue experiment - not enough data or not significant"
        
        results = experiment['results']
        variants = list(results.keys())
        
        if results[variants[0]]['conversion_rate'] > results[variants[1]]['conversion_rate']:
            return f"Implement {variants[0]} - higher conversion rate"
        else:
            return f"Implement {variants[1]} - higher conversion rate"

import pandas as pd
import numpy as np

class ConversionRateOptimization:
    def __init__(self):
        self.funnel_steps = {}
        self.conversion_data = {}
        self.optimization_recommendations = {}
    
    def analyze_funnel(self, funnel_data):
        """Analizar embudo de conversi√≥n"""
        funnel_analysis = {}
        
        for step, data in funnel_data.items():
            funnel_analysis[step] = {
                'visitors': len(data),
                'conversions': sum(data['converted']),
                'conversion_rate': sum(data['converted']) / len(data),
                'drop_off_rate': 1 - (sum(data['converted']) / len(data))
            }
        
        return funnel_analysis
    
    def identify_bottlenecks(self, funnel_analysis):
        """Identificar cuellos de botella"""
        bottlenecks = []
        
        for step, analysis in funnel_analysis.items():
            if analysis['drop_off_rate'] > 0.5:  # M√°s del 50% de abandono
                bottlenecks.append({
                    'step': step,
                    'drop_off_rate': analysis['drop_off_rate'],
                    'priority': 'high' if analysis['drop_off_rate'] > 0.7 else 'medium'
                })
        
        return sorted(bottlenecks, key=lambda x: x['drop_off_rate'], reverse=True)
    
    def generate_optimization_recommendations(self, bottlenecks, user_behavior_data):
