def analyze_query_quality(queries):
    """Analiza calidad de consultas"""
    quality_scores = []
    
    for query in queries:
        score = 0
        
        # Longitud (20-100 palabras es ideal)
        word_count = len(query.split())
        if 20 <= word_count <= 100:
            score += 0.3
        elif 10 <= word_count < 20 or 100 < word_count <= 150:
            score += 0.15
        
        # Especificidad (palabras clave espec√≠ficas)
        specific_keywords = ['contrato', 'informe', 'propuesta', 'an√°lisis']
        if any(kw in query.lower() for kw in specific_keywords):
            score += 0.3
        
        # Contexto (informaci√≥n adicional)
        context_indicators = ['para', 'incluyendo', 'con', 'seg√∫n']
        if any(ind in query.lower() for ind in context_indicators):
            score += 0.2
        
        # Claridad (estructura clara)
        if '?' not in query and len(query.split('.')) <= 3:
            score += 0.2
        
        quality_scores.append(score)
    
    return np.mean(quality_scores)
