class SpaceMiningAnalyzer:
    def __init__(self, celestial_bodies, mining_technologies):
        self.celestial_bodies = celestial_bodies
        self.mining_technologies = mining_technologies
        self.resource_estimator = ResourceEstimator()
        self.mining_cost_calculator = MiningCostCalculator()
        self.space_logistics = SpaceLogistics()
    
    def analyze_mining_opportunity(self, celestial_body, resource_type):
        # Estimate resource availability
        resource_estimate = self.resource_estimator.estimate_resources(celestial_body, resource_type)
        
        # Calculate mining costs
        mining_costs = self.mining_cost_calculator.calculate_costs(
            celestial_body, resource_type, self.mining_technologies
        )
        
        # Calculate logistics costs
        logistics_costs = self.space_logistics.calculate_logistics_costs(
            celestial_body, resource_type
        )
        
        # Calculate profitability
        profitability = self.calculate_profitability(resource_estimate, mining_costs, logistics_costs)
        
        return SpaceMiningAnalysis(resource_estimate, mining_costs, logistics_costs, profitability)
    
    def calculate_profitability(self, resource_estimate, mining_costs, logistics_costs):
        # Calculate net present value
        total_costs = mining_costs + logistics_costs
        npv = resource_estimate - total_costs
        
        # Calculate return on investment
        roi = npv / total_costs if total_costs > 0 else 0
        
        return ProfitabilityAnalysis(npv, roi)
