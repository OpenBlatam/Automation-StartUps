```python
# ROI Calculation Formula
def calculate_roi(investment, benefits, years):
    total_investment = investment * years
    total_benefits = benefits * years
    net_benefit = total_benefits - total_investment
    roi_percentage = (net_benefit / total_investment) * 100
    payback_period = investment / (benefits / 12)  # months
    
    return {
        'total_investment': total_investment,
        'total_benefits': total_benefits,
        'net_benefit': net_benefit,
        'roi_percentage': roi_percentage,
        'payback_period_months': payback_period
    }

# Example calculation
investment = 1000000  # $1M initial investment
annual_benefits = 5000000  # $5M annual benefits
years = 3

result = calculate_roi(investment, annual_benefits, years)
print(f"3-Year ROI: {result['roi_percentage']:.1f}%")
print(f"Payback Period: {result['payback_period_months']:.1f} months")
```
