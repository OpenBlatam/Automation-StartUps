const HRISIntegration = {
    syncEmployeeData: async function() {

                'Authorization': 'Bearer ' + apiToken,
                'Content-Type': 'application/json'

            method: 'PUT',
            headers: {
                'Authorization': 'Bearer ' + apiToken,
                'Content-Type': 'application/json'
            },
            body: JSON.stringify({ status: status })
        });
    }
};
