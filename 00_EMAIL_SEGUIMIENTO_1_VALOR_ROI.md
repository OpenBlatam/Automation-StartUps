def calculate_roi(horas_semanales, tarifa_hora, tareas_automatizables):
    """
    Calcula ROI personalizado para prospecto
    
    Args:
        horas_semanales: Horas que pasa en tareas automatizables
        tarifa_hora: Tarifa por hora del prospecto
        tareas_automatizables: % de tareas que se pueden automatizar
    
    Returns:
        dict con c√°lculos de ROI
    """
    horas_mes = horas_semanales * 4

    # Con IA: 70% de reducci√≥n

    ahorro_mensual = costo_actual - costo_ia
    ahorro_anual = ahorro_mensual * 12
    
    return {
        'horas_actuales_mes': horas_automatizables,
        'costo_actual_mes': costo_actual,
        'horas_con_ia_mes': horas_ia,
        'costo_con_ia_mes': costo_ia,
        'ahorro_mensual': ahorro_mensual,
        'ahorro_anual': ahorro_anual,
        'roi_percentage': (ahorro_anual / 1200) * 100  # Asumiendo precio $100/mes
    }

# Ejemplo de uso
