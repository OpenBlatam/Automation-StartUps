def optimize_cac_by_channel(campaign_data):
    """
    Optimiza presupuesto entre canales para minimizar CAC
    """
    results = []
    
    for channel in campaign_data['channel'].unique():
        channel_data = campaign_data[campaign_data['channel'] == channel]
        
        total_spend = channel_data['spend'].sum()
        total_leads = channel_data['leads'].sum()
        total_trials = channel_data['trials'].sum()
        total_revenue = channel_data['revenue'].sum()
        
        cac = total_spend / total_leads if total_leads > 0 else 0
        trial_cac = total_spend / total_trials if total_trials > 0 else 0
        roi = (total_revenue - total_spend) / total_spend if total_spend > 0 else 0
        
        results.append({
            'channel': channel,
            'spend': total_spend,
            'leads': total_leads,
            'trials': total_trials,
            'revenue': total_revenue,
            'cac': cac,
            'trial_cac': trial_cac,
            'roi': roi * 100
        })
    
    results_df = pd.DataFrame(results)
    results_df = results_df.sort_values('roi', ascending=False)
    
    # RecomendaciÃ³n de presupuesto
    total_budget = campaign_data['spend'].sum()
    optimal_allocation = {}
    
    for _, row in results_df.iterrows():
        if row['roi'] > 100:  # ROI positivo
            allocation = (row['roi'] / results_df['roi'].sum()) * total_budget
            optimal_allocation[row['channel']] = allocation
    
    return results_df, optimal_allocation

    """

    # PredicciÃ³n
    forecast = []
    for month in range(1, months_ahead + 1):
        # Asumir crecimiento del 10% mensual (ajustar segÃºn datos)

            'predicted_trials': predicted_trials,

        route_to('mid_market_team')
        send_demo_invitation(lead_data)
    else:
        route_to('nurture_sequence')
        send_educational_content(lead_data)
    
    # 3. Nurture automÃ¡tico
    if lead_data['stage'] == 'nurture':
        send_automated_nurture_sequence(lead_data)
    
    # 4. Follow-up automÃ¡tico
    if lead_data['demo_attended'] and not lead_data['trial_started']:
        send_trial_invitation(lead_data)
    
    # 5. Churn prevention automÃ¡tico
    if detect_churn_risk(lead_data):
        send_churn_prevention_sequence(lead_data)
    
    return {
        'lead_score': score,
        'routing': route_to,
        'next_actions': get_next_actions(lead_data)
    }
