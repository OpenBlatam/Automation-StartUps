```python
import pandas as pd
import numpy as np
from scipy.optimize import minimize

def optimizar_presupuesto_multicanal(datos_rendimiento, presupuesto_total):
    """
    Optimiza distribuci√≥n de presupuesto entre canales usando algoritmos de optimizaci√≥n
    """
    
    # Preparar datos
    canales = datos_rendimiento['canal'].unique()
    
    # Calcular ROI hist√≥rico por canal
    rois = {}
    for canal in canales:
        datos_canal = datos_rendimiento[datos_rendimiento['canal'] == canal]
        roi = datos_canal['ingresos'].sum() / datos_canal['costo'].sum()
        rois[canal] = roi
    
    # Funci√≥n objetivo: maximizar ROI total
    def objetivo(distribucion):
        roi_total = 0
        for i, canal in enumerate(canales):
            roi_total += distribucion[i] * rois[canal]
        return -roi_total  # Negativo porque minimize busca m√≠nimo
    
    # Restricciones
    def restriccion_presupuesto(distribucion):
        return presupuesto_total - sum(distribucion)
    
    # Restricci√≥n: m√≠nimo 10% por canal
    def restriccion_minimo(distribucion):
        return distribucion - (presupuesto_total * 0.1)
    
    # Valores iniciales (distribuci√≥n equitativa)
    x0 = [presupuesto_total / len(canales)] * len(canales)
    
    # Optimizar
    resultado = minimize(
        objetivo,
        x0,
        method='SLSQP',
        constraints=[
            {'type': 'eq', 'fun': restriccion_presupuesto},
            {'type': 'ineq', 'fun': restriccion_minimo}
        ],
        bounds=[(0, presupuesto_total)] * len(canales)
    )
    
    # Formatear resultado
    distribucion_optimizada = {}
    for i, canal in enumerate(canales):
        distribucion_optimizada[canal] = {
            'presupuesto': resultado.x[i],
            'porcentaje': (resultado.x[i] / presupuesto_total) * 100,
            'roi_esperado': rois[canal]
        }
    
    return distribucion_optimizada

# Ejemplo de uso

    'costo': [4000, 3500, 1500, 1000],

})
```

```python
import requests
import json
from typing import Dict, List

class HubSpotIntegration:
    def __init__(self, api_key: str):
        self.api_key = api_key
        self.base_url = "https://api.hubapi.com"
        self.headers = {
            "Authorization": f"Bearer {api_key}",
            "Content-Type": "application/json"
        }
    
    def crear_contacto(self, datos: Dict) -> Dict:
        """Crea contacto en HubSpot"""
        url = f"{self.base_url}/crm/v3/objects/contacts"
        
        properties = {
            "email": datos.get("email"),
            "firstname": datos.get("nombre"),
            "lastname": datos.get("apellido"),
            "company": datos.get("empresa"),
            "phone": datos.get("telefono"),
            "industry": datos.get("industria"),
            "lifecyclestage": "customer"
        }
        
        payload = {"properties": properties}
        
        response = requests.post(url, headers=self.headers, json=payload)
        response.raise_for_status()
        
        return response.json()
    
    def crear_deal(self, contacto_id: str, deal_data: Dict) -> Dict:
        """Crea deal asociado a contacto"""
        url = f"{self.base_url}/crm/v3/objects/deals"
        
        payload = {
            "properties": {
                "dealname": deal_data.get("nombre"),
                "amount": deal_data.get("valor"),
                "dealstage": "appointmentscheduled",
                "pipeline": "default"
            },
            "associations": [
                {
                    "to": {"id": contacto_id},
                    "types": [{"associationCategory": "HUBSPOT_DEFINED", 
                              "associationTypeId": 3}]
                }
            ]
        }
        
        response = requests.post(url, headers=self.headers, json=payload)
        response.raise_for_status()
        
        return response.json()
    
    def sincronizar_audiencia(self, segmento: str, contactos: List[Dict]):
        """Sincroniza audiencia para campa√±as"""
        # Crear lista est√°tica en HubSpot
        url = f"{self.base_url}/contacts/v1/lists"
        
        payload = {
            "name": f"Audiencia {segmento}",
            "dynamic": False
        }
        
        response = requests.post(url, headers=self.headers, json=payload)
        lista_id = response.json()["listId"]
        
        # Agregar contactos
        url_add = f"{self.base_url}/contacts/v1/lists/{lista_id}/add"
        
        emails = [c["email"] for c in contactos]
        payload = {"emails": emails}
        
        requests.post(url_add, headers=self.headers, json=payload)
        
        return lista_id
```

### Calculadora de ROI por Canal

```python
class CalculadoraROI:
    """Calcula ROI y optimiza presupuesto de marketing"""
    
    def calcular_roi_canal(self, canal: str, datos: dict) -> float:
        """Calcula ROI de un canal espec√≠fico"""
        ingresos = datos.get("ingresos", 0)
        costo = datos.get("costo", 0)
        
        if costo == 0:
            return float('inf')
        
        roi = (ingresos - costo) / costo
        return roi
    
    def optimizar_presupuesto(self, canales: dict, presupuesto_total: float) -> dict:
        """Optimiza distribuci√≥n de presupuesto"""
        # Calcular ROI por canal
        rois = {canal: self.calcular_roi_canal(canal, datos) 
                for canal, datos in canales.items()}
        
        # Ordenar por ROI
        canales_ordenados = sorted(rois.items(), key=lambda x: x[1], reverse=True)
        
        # Distribuir presupuesto proporcionalmente al ROI
        distribucion = {}
        total_roi = sum(rois.values())
        
        for canal, roi in canales_ordenados:
            porcentaje = roi / total_roi if total_roi > 0 else 1 / len(canales)
            distribucion[canal] = {
                "presupuesto": presupuesto_total * porcentaje,
                "roi": roi,
                "porcentaje": porcentaje * 100
            }
        
        return distribucion
```

```python
from sklearn.ensemble import RandomForestClassifier
import pandas as pd

class AnalizadorPredictivo:
    """Analiza y predice performance de campa√±as"""
    
    def __init__(self):
        self.modelo = RandomForestClassifier(n_estimators=100)
        self.entrenado = False
    
    def entrenar(self, datos_historicos: pd.DataFrame):
        """Entrena modelo con datos hist√≥ricos"""
        X = datos_historicos[['presupuesto', 'canal', 'audiencia_size', 'copy_length']]
        y = datos_historicos['conversion_rate']
        
        self.modelo.fit(X, y)
        self.entrenado = True
    
    def predecir_conversion(self, campana: dict) -> float:
        """Predice tasa de conversi√≥n"""
        if not self.entrenado:
            raise Exception("Modelo no entrenado")
        
        X = pd.DataFrame([{
            'presupuesto': campana['presupuesto'],
            'canal': campana['canal'],
            'audiencia_size': campana['audiencia_size'],
            'copy_length': len(campana['copy'])
        }])
        
        prediccion = self.modelo.predict_proba(X)[0][1]
        return prediccion
    
    def recomendar_optimizaciones(self, campana: dict) -> List[dict]:
        """Recomienda optimizaciones basadas en predicci√≥n"""
        conversion_actual = self.predecir_conversion(campana)
        recomendaciones = []
        
        # Probar diferentes configuraciones
        configuraciones = [
            {"presupuesto": campana['presupuesto'] * 1.2},
            {"audiencia_size": campana['audiencia_size'] * 0.8},
            {"copy_length": len(campana['copy']) * 1.3}
        ]
        
        for config in configuraciones:
            campana_test = {**campana, **config}
            conversion_test = self.predecir_conversion(campana_test)
            
            if conversion_test > conversion_actual * 1.1:
                recomendaciones.append({
                    "configuracion": config,
                    "mejora_esperada": (conversion_test - conversion_actual) / conversion_actual
                })
        
        return sorted(recomendaciones, key=lambda x: x['mejora_esperada'], reverse=True)
```
