### **1. Script Python: Calculadora de M√©tricas**

```python
#!/usr/bin/env python3
"""
Calculadora de M√©tricas para Startups
Calcula CAC, LTV, LTV/CAC, Payback Period, y m√°s
"""

def calculate_cac(marketing_cost, sales_cost, new_customers):
    """Calcula Customer Acquisition Cost"""
    if new_customers == 0:
        return 0
    return (marketing_cost + sales_cost) / new_customers

def calculate_ltv(arpu, gross_margin, churn_rate):
    """Calcula Lifetime Value"""
    if churn_rate == 0:
        return float('inf')
    return (arpu * gross_margin) / churn_rate

def calculate_ltv_cac_ratio(ltv, cac):
    """Calcula ratio LTV/CAC"""
    if cac == 0:
        return float('inf')
    return ltv / cac

def calculate_payback_period(cac, arpu, gross_margin):
    """Calcula Payback Period en meses"""
    if arpu * gross_margin == 0:
        return float('inf')
    return cac / (arpu * gross_margin)

def calculate_viral_coefficient(invites_per_user, conversion_rate):
    """Calcula Viral Coefficient (K-Factor)"""
    return invites_per_user * conversion_rate

# Ejemplo de uso
if __name__ == "__main__":
    # Inputs
    marketing_cost = 10000
    sales_cost = 5000
    new_customers = 50
    arpu = 100  # Average Revenue Per User (mensual)
    gross_margin = 0.80  # 80%
    churn_rate = 0.05  # 5% mensual
    
    # C√°lculos
    cac = calculate_cac(marketing_cost, sales_cost, new_customers)
    ltv = calculate_ltv(arpu, gross_margin, churn_rate)
    ltv_cac = calculate_ltv_cac_ratio(ltv, cac)
    payback = calculate_payback_period(cac, arpu, gross_margin)
    
    # Output
    print(f"CAC: ${cac:.2f}")
    print(f"LTV: ${ltv:.2f}")
    print(f"LTV/CAC: {ltv_cac:.2f}:1")
    print(f"Payback Period: {payback:.1f} meses")
    
    # An√°lisis
    if ltv_cac >= 3:
        print("‚úÖ Unit economics saludables")
    else:
```

```python
#!/usr/bin/env python3
"""
Analizador de Cohortes para Startups
Analiza retenci√≥n por cohorte de usuarios
"""

import pandas as pd
from datetime import datetime, timedelta

def analyze_cohorts(df, signup_date_col, user_id_col, activity_date_col):
    """
    Analiza retenci√≥n por cohorte
    
    Args:
        df: DataFrame con datos de usuarios
        signup_date_col: Columna con fecha de signup
        user_id_col: Columna con ID de usuario
        activity_date_col: Columna con fecha de actividad
    """
    # Crear cohortes mensuales
    df['cohort'] = pd.to_datetime(df[signup_date_col]).dt.to_period('M')
    df['period'] = pd.to_datetime(df[activity_date_col]).dt.to_period('M')
    
    # Calcular per√≠odo desde signup
    df['period_number'] = (df['period'] - df['cohort']).apply(attrgetter('n'))
    
    # Agrupar por cohorte y per√≠odo
    cohort_data = df.groupby(['cohort', 'period_number'])[user_id_col].nunique().reset_index()
    cohort_pivot = cohort_data.pivot(index='cohort', columns='period_number', values=user_id_col)
    
    # Calcular retenci√≥n
    cohort_size = cohort_pivot.iloc[:, 0]
    retention = cohort_pivot.divide(cohort_size, axis=0)
    
    return retention
```
