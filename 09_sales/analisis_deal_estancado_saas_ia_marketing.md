```python
"""
Sistema de Reactivaci√≥n de Trials Estancados - SaaS
Versi√≥n: 1.0

import pandas as pd
from datetime import datetime, timedelta
import requests

class TrialReactivationSystem:
    def __init__(self, product_api_key, email_api_key):
        self.product_api_key = product_api_key
        self.email_api_key = email_api_key
        
    def get_stagnant_trials(self, days_threshold=2):
        """Obtiene trials estancados"""
        url = f"https://api.product.com/trials?stagnant={days_threshold}"
        headers = {"Authorization": f"Bearer {self.product_api_key}"}
        response = requests.get(url, headers=headers)
        return response.json()
    
    def calculate_product_health_score(self, trial_data):
        """Calcula health score basado en uso del producto"""
        trial_status = 100 if trial_data['activated'] else 0
        usage_score = (trial_data['actions_count'] / trial_data['expected_actions']) * 100
        engagement_score = trial_data['email_interactions'] * 10
        fit_score = trial_data['feature_match'] * 100
        
        health_score = (
            trial_status * 0.4 +
            min(usage_score, 100) * 0.3 +
            min(engagement_score, 100) * 0.2 +
            fit_score * 0.1
        )
        
        return health_score
    
    def send_in_app_notification(self, user_id, message):
        """Env√≠a notificaci√≥n in-app"""
        url = "https://api.product.com/notifications"
        headers = {"Authorization": f"Bearer {self.product_api_key}"}
        data = {
            "user_id": user_id,
            "message": message,
            "type": "reactivation"
        }
        return requests.post(url, headers=headers, json=data)
    
    def process_trials(self):
        """Procesa todos los trials estancados"""
        trials = self.get_stagnant_trials()
        results = []
        
        for trial in trials:
            health_score = self.calculate_product_health_score(trial)
            
            if not trial['activated']:
                # Notificaci√≥n in-app
                self.send_in_app_notification(
                    trial['user_id'],
                    "¬°Activa tu trial ahora y genera tu primer copy en 30 segundos!"
                )
                strategy = "Estrategia 4 - Product-Led"
            elif health_score > 70:
                strategy = "Estrategia 3 - Urgencia"
            elif health_score > 50:
                strategy = "Estrategia 1 - Valor"
            else:
                strategy = "Estrategia 5 - Educativo"
            
            results.append({
                'trial_id': trial['id'],
                'health_score': health_score,
                'strategy': strategy
            })
        
        return results
```
